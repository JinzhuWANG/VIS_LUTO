import altair as alt

alt.Chart(crosstab_lu_all.reset_index()).mark_rect().encode(
    x='Land-use from:N',
    y='Land-use to:N',
    color='Area (km2):Q',
    tooltip=['Land-use from','Land-use to','Area (km2)']
).properties(
    width=400,
    height=400
).interactive()
